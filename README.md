// PP_CLIENT_ID: The active PayPal ClientId

// PP_SANDBOX_LIVE: The environment to use: Sandbox | Live
// PP_PARTNER_ID: PayPal Partner ID given by account manager (if available)
// STRIPE_SECRET: The active stripe secret
